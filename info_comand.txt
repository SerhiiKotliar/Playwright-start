//запускает тесты и отчёты по ним в папку
python -m pytest -v -s --alluredir result
//открывает в браузере отчёт
allure serve result
//генерирует файлы отчёта в папку
allure generate report
//открывает сохранённый отчёт в папке report
allure open report
///////////////////////////////////////////////////////////////
//Как использовать файл allure_runner.py
//Запуск всех тестов
python allure_runner.py tests
//Открыть отчёт напрямую через allure serve
python allure_runner.py serve
//Сгенерировать отчёт и открыть в браузере (Python-сервер)
python allure_runner.py
//Запустить всё сразу (тесты → отчёт → сервер)
python allure_runner.py all
